  def read(filename) 
    @occurrences = Hash.new { |h, k| h[k] = 0 }
    File.open(filename, "r") { |f|
          f.each_line { |line|
            words = line.split
            words.each { |w|
              if ! hide_list.include?(w.downcase)
                @occurrences[w.downcase] += 1 
              end
            }
          }
        }
    @occurrences
  end
